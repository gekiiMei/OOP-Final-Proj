CREATE TABLE finals.GRADE (
    sy VARCHAR(10) NOT NULL,
    semester VARCHAR(1) NOT NULL,
    student_no INT NOT NULL,
    subject_code VARCHAR(50) NOT NULL,
    block_no VARCHAR(10),
    grade DECIMAL(4, 2),
    PRIMARY KEY (sy, semester, student_no, subject_code),
    FOREIGN KEY (sy, semester, student_no, subject_code) 
        REFERENCES SUBJECT_SCHEDULE(sy, semester, student_no, subject_code)
);
