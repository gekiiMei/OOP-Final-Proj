CREATE TABLE finals.PASSWORD (
    user_id VARCHAR(20) NOT NULL,
    password VARCHAR(255) NOT NULL
);

--vv== JAVA CODE, PLACE IN A CLASS, IMPORT JAVA.SQL.*, RUN, THEN DELETE CLASS ==vv--
    public static void main(String[] args) throws SQLException {
        Connection con = ConnectDB.connect();
        ResultSet rs = con.prepareStatement("SELECT student_no FROM finals.STUDENT").executeQuery();
        while (rs.next()) {
            con.prepareStatement("INSERT INTO finals.PASSWORD VALUES ('" + rs.getString("student_no") + "', '"+ rs.getString("student_no") + "')").execute();
        }
        
        rs = con.prepareStatement("SELECT employee_id FROM finals.EMPLOYEE").executeQuery();
        while (rs.next()) {
            con.prepareStatement("INSERT INTO finals.PASSWORD VALUES ('" + rs.getString("employee_id") + "', '"+ rs.getString("employee_id") + "')").execute();
        }
    }
