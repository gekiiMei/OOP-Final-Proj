CREATE TABLE finals.college (
    college_code    VARCHAR(25)     NOT NULL
   ,description     VARCHAR(100)    NOT NULL
   ,date_opened     VARCHAR(30)     NOT NULL
   ,date_closed     VARCHAR(30)     NOT NULL
   ,status          VARCHAR(1)      NOT NULL
   ,constraint college_pk PRIMARY KEY(college_code, date_opened, date_closed)
   ,constraint college_code_uk UNIQUE (college_code)
)
