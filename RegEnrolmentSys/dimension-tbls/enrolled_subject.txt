CREATE TABLE finals.ENROLLED_SUBJECT (
    sy INT NOT NULL,
    semester INT NOT NULL,
    student_no INT NOT NULL,
    subject_code VARCHAR(50) NOT NULL,
    block_no VARCHAR(50) NOT NULL,
    status VARCHAR(1) DEFAULT 'N',
    PRIMARY KEY (sy, semester, student_no, subject_code, block_no),
    FOREIGN KEY (sy, semester, student_no, subject_code, block_no)
        REFERENCES SUBJECT_SCHEDULE(sy, semester, student_no, subject_code)
);
